 21 /**
 22  * This regular expression tests for and separates the `word` from it's 
 23  *  syntactic marker.
 24  * 
 25  * The following notes are from wninput(5WN) at
 26  * http://wordnet.princeton.edu/man/wninput.5WN.html
 27  * 
 28  * ASCII form of a word as entered in the synset by the lexicographer, with 
 29  * spaces replaced by underscore characters (_). The text of the word is 
 30  * case sensitive, in contrast to its form in the corresponding index. pos 
 31  * file, that contains only lower-case forms. In data.adj , a word is 
 32  * followed by a syntactic marker if one was specified in the lexicographer 
 33  * file. A syntactic marker is appended, in parentheses, onto word without 
 34  * any intervening spaces. See wninput(5WN) for a list of the syntactic 
 35  * markers for adjectives.
 36  * 
 37  * The syntactic markers are:
 38  * 
 39  * (p)    predicate position
 40  * (a)    prenominal (attributive) position
 41  * (ip)   immediately postnominal position
 42  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 43  * @version 20130521
 44  * @private
 45  */
 46 var syntactic_markers = /(.*)\((..?)\)/;
 47 /**
 48  * This regular expression separates the source and target identifiers.
 49  * this pattern returns ["0a3b", "0a", "3b"]
 50  * for "0a3b".match(source_target);
 51  *  
 52  * The following notes are from
 53  * http://wordnet.princeton.edu/man/wninput.5WN.html
 54  *  
 55  * The source/target field distinguishes lexical and semantic pointers. It 
 56  * is a four byte field, containing two two-digit hexadecimal integers. The 
 57  * first two digits indicates the word number in the current (source) 
 58  * synset, the last two digits indicate the word number in the target 
 59  * synset. A value of 0000 means that pointer_symbol represents a semantic 
 60  * relation between the current (source) synset and the target synset 
 61  * indicated by synset_offset.
 62  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 63  * @version 20130521
 64  * @private
 65  */
 66 var source_target = /([0-9a-f]{2})([0-9a-f]{2})/;
 67 /**
 68  * Separates the license from the given fileContents and feeds the remaining 
 69  *  lines into the given entry parser.
 70  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 71  * @version 20130521
 72  * @private
 73  * @param {String} fileContents The file contents to parse.
 74  * @param {Function} entryParser The entry parser to use.
 75  * @returns {Object} Returns an object representing the parsed file.
 76  */
 77 function wordNetFileToArray (fileContents, entryParser) {
 78     'use strict';
 79     var comments = /^ {2}.*$/gm;
 80     function getLicense () {
 81         var out = '';
 82         try {
 83             out = fileContents.match(comments).join('\n');
 84         } catch (e) {
 85             out = '';
 86         }
 87         return out;
 88     }
 89     var out = {
 90         "license" : getLicense(),
 91         "entries" : fileContents.replace(comments, '')
 92             .replace(/(\r\n|\r|\n)+/g, '\n')
 93             .trim()
 94             .split('\n')
 95             .map(function (entry) {
 96                 return entryParser(entry);
 97             })
 98     };
 99     return out;
100 }

349 function parseDataFileFormat (record) {
350     'use strict';
351     var out = {};
352     var tmp = record.split('|');
353     out.gloss = tmp[1].trim();
354     tmp = tmp[0].trim().split(' ');
355     out.synset_offset = tmp.shift();
356     out.lex_filenum = tmp.shift();
357     out.ss_type = tmp.shift();
358     out.w_cnt = tmp.shift();
359     var numWords = parseInt(out.w_cnt, 16);
360     out.words = [];
361     var wrd = '';
362     var synMark = '';
363     while (numWords > 0) {
364         wrd = tmp.shift();
365         synMark = '';
366         if (syntactic_markers.test(wrd)) {
367             synMark = wrd.match(syntactic_markers)[2];
368             wrd = wrd.match(syntactic_markers)[1];
369         }
370         out.words.push({
371             "word" : wrd,
372             "syntactic_markers" : synMark,
373             "lex_id" : tmp.shift()
374         });
375         numWords -= 1;
376     }
377     out.p_cnt = tmp.shift();
378     var numPtr = parseInt(out.p_cnt, 10);
379     var drawer = '';
380     out.pointers = [];
381     while (numPtr > 0) {
382         drawer = {
383             "pointer_symbol" : tmp.shift(),
384             "synset_offset" : tmp.shift(),
385             "pos" : tmp.shift(),
386             "source_target" : tmp.shift()
387         };
388         drawer.source = drawer.source_target.match(source_target)[1];
389         drawer.target = drawer.source_target.match(source_target)[2];
390         out.pointers.push(drawer);
391         numPtr -= 1;
392     }
393     out.f_cnt = 0;
394     out.frames = [];
395     if (tmp.length > 0) {
396         out.f_cnt = tmp.shift();
397         var numFrames = parseInt(out.f_cnt, 10);
398         while (numFrames > 0) {
399             tmp.shift(); // throw away superfluous + sign, there's a count.
400             out.frames.push({
401                 "f_num" : tmp.shift(), 
402                 "w_num" : tmp.shift()
403             });
404             numFrames -= 1;
405         }
406     }
407     return out;
408 }
409 /**
410  * Parses the `index` file's individual records.
411  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
412  * @version 20130521
413  * @private
414  * @param {String} record A line of information from the `index` file.
415  * @returns {Object} Returns an object containing properties whose values are 
416  *  determined by the parsed data.
417  * @example
418  * // This is example output
419  * 
420  * { lemma: '\'tween',
421  *   pos: 'r',
422  *   synset_cnt: '1',
423  *   p_cnt: '0',
424  *   pointers: [],
425  *   sense_cnt: '1',
426  *   tagsense_cnt: '0',
427  *   synset_offsets: [ '00250898' ] }
428  * 
429  * @example
430  * // This is example output
431  * 
432  * { lemma: 'aah',
433  *   pos: 'v',
434  *   synset_cnt: '1',
435  *   p_cnt: '1',
436  *   pointers: [ { ptr_symbol: '@' } ],
437  *   sense_cnt: '1',
438  *   tagsense_cnt: '0',
439  *   synset_offsets: [ '00865776' ] }
440  * 
441  * @example
442  * // This is example output
443  * 
444  * { lemma: '.22-caliber',
445  *   pos: 'a',
446  *   synset_cnt: '1',
447  *   p_cnt: '1',
448  *   pointers: [ { ptr_symbol: '\\' } ],
449  *   sense_cnt: '1',
450  *   tagsense_cnt: '0',
451  *   synset_offsets: [ '03146310' ] }
452  * 
453  * @example
454  * // This is example output
455  * 
456  * { lemma: '1000000000000',
457  *   pos: 'n',
458  *   synset_cnt: '2',
459  *   p_cnt: '2',
460  *   pointers: [ { ptr_symbol: '@' }, { ptr_symbol: ';' } ],
461  *   sense_cnt: '2',
462  *   tagsense_cnt: '0',
463  *   synset_offsets: [ '13752443', '13752172' ] }
464  */
465 function parseIndexFileFormat (record) {
466     'use strict';
467     var out = {};
468     var tmp = record.trim().split(' ');
469     out.lemma = tmp.shift();
470     out.pos = tmp.shift();
471     out.synset_cnt = tmp.shift();
472     out.p_cnt = tmp.shift();
473     var numPtr = out.p_cnt;
474     out.pointers = [];
475     while (numPtr > 0) {
476         out.pointers.push({
477             "ptr_symbol" : tmp.shift()
478         });
479         numPtr -= 1;
480     }
481     out.sense_cnt = tmp.shift();
482     out.tagsense_cnt = tmp.shift();
483     out.synset_offsets = [];
484     var numSynsets = out.synset_cnt;
485     while(numSynsets > 0) {
486         out.synset_offsets.push(tmp.shift());
487         numSynsets -= 1;
488     }
489     return out;
490 }

569 /**
570  * Converts `data` and `index` files into arrays.
571  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
572  * @version 20130521
573  * @param {String} wnDirectory The directory containing the wordnet database 
574  *  files.
575  * @param {Function} callback A callback function which will be given the 
576  *  name of the file and each array as it is generated.
577  */
578 function wordnetToArrays (wnDirectory, callback) {
579     var fs = require('fs');
580     var path = require('path');
581     var wordNetDatabaseFiles = [
582         "index.noun",
583         "data.noun",
584         "index.verb",
585         "data.verb",
586         "index.adj",
587         "data.adj",
588         "index.adv",
589         "data.adv"
590     ];
591     wordNetDatabaseFiles.forEach(function (filename) {
592         'use strict';
593         var fname = path.resolve(wnDirectory, filename);
594         
595         fs.readFile(fname, 'utf8', function (err, contents) {
596             if (err) {
597                 throw err;
598             }
599             if (/data/.test(filename)) {
600                 callback(filename, wordNetDataFileToArray(contents));
601             }
602             if (/index/.test(filename)) {
603                 callback(filename, wordNetIndexFileToArray(contents));
604             }
605         });
606     });
607 }
608 /**
609  * Will convert `data` and `index` files into an SQLite database at some future
610  *  date. Right now all it does is log the first entry from each of the 
611  *  generated arrays along with the parsed file's name.
612  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
613  * @version 20130521
614  */
615 function wordnetToSqliteDb () {
616     var wordnet = require('WNdb');
617     wordnetToArrays(wordnet.path, function (filename, arr) {
618         console.log(filename + ' =', arr.entries[0]); 
619     });
620 }
